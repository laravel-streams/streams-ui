(function () {
    var streams = window.streams
    var app = streams.core.app;

    app
        .initialize({
            providers: [
                streams.core.CoreServiceProvider,
                streams.ui.UiServiceProvider,
                window.app.AppServiceProvider
            ],
            config   : {
                streams: {},
                http   : {}
            }
        })
        .then(app.boot.bind(app))
        .then(function (app) {
            app.start();
        })
        .then(function (app) {
            // started
        });
}).call(window);
</script>
